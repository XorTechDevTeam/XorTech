45 XtModRC XtModuleManager::addModule(XtModule *module) {
46  LOGMSG("[XtModuleManager]: linking module %s...", module->getName().c_str());
47  module->setId(this->modules.size());
48  this->modules.push_back(module);
49 
50  XtDependencyUnit *deps = module->getDependencies();
51 
52  auto cur = deps->nextDependency();
53 
54  int dep;
55  while (cur != deps->end()) {
56  dep = 0;
57 
58  for (int i = 0; i < (int)this->modules.size(); ++i) {
59  if (this->modules.at(i)->getName() == cur->first) {
60  *(cur->second) = dep = i;
61  }
62  }
63 
64  if (dep == 0) {
65  LOGMSG("[XtModuleManager]: unable to resolve dependency '%s'.", cur->first.c_str());
66  return RC_UNRSLV_DEP;
67  }
68  }
69 
70  return RC_OK;
71 }
